Promise.all guarantee the order of response in input promises so that caller can retrieve the result with array index number.

[code lang="c"]
var allDone = Promise.all([
promise1,
promise2
])
.then(res => {
res[0]; // promise1 result
res[1]; // promise2 result
});
[/code]
