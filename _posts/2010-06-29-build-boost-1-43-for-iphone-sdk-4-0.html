1. modify user.config.jam in tools/build/v2 directory
{% highlight wl linenos %} using darwin : 4.2.1~iphone : /Developer/Platforms/iPhoneOS.platform/Developer/usr/bin/g++ -arch armv7 : : arm iphone ; using darwin : 4.2.1~iphonesim : /Developer/Platforms/iPhoneSimulator.platform/Developer/usr/bin/g++ : : x86 iphone ; {% endhighlight %}

2. compile
{% highlight wl linenos %} ./bootstrap.sh ./bjam --prefix=${HOME}/usr/boost-1.43/iphone toolset=darwin architecture=arm target-os=iphone macosx-version=iphone-4.0 define=_LITTLE_ENDIAN link=static install {% endhighlight %}
