Syntax: => A

[code land="c"]
def ifElse[A](test: Boolean, whenTrue: => A, whenFalse: => A): A = test match {
case true => whenTrue
case false => whenFalse
}

ifElse(1 == 1, println("true"), println("false"))
[/code]

By Name parameter will be evaluated when called
